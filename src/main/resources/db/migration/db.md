- auth_identities
```sql
create table auth_identities
(
    id               bigserial
        primary key,
    user_id          bigint                                 not null
        constraint fk_auth_identities_user
            references users,
    provider         varchar(20)                            not null,
    provider_user_id varchar(200)                           not null,
    email            varchar(200),
    created_at       timestamp with time zone default now() not null,
    updated_at       timestamp with time zone default now() not null
);

alter table auth_identities
    owner to verlnn;

create unique index uq_auth_identities_provider_user
    on auth_identities (provider, provider_user_id);

create index idx_auth_identities_user_id
    on auth_identities (user_id);

create unique index uk_auth_id_provider_user
    on auth_identities (provider, provider_user_id);

create unique index uk_auth_id_email
    on auth_identities (email)
    where (email IS NOT NULL);

create trigger trg_auth_identities_updated_at
    before update
    on auth_identities
    for each row
execute procedure touch_updated_at();
```

- email_verifications
```sql
create table email_verifications
(
    id                  bigint generated by default as identity
        primary key,
    user_id             bigint                                 not null
        references users,
    email               varchar(320)                           not null,
    code_hash           varchar(255)                           not null,
    expires_at          timestamp with time zone               not null,
    verified_at         timestamp with time zone,
    attempt_count       integer                  default 0     not null
        constraint chk_ev_attempt_count
            check (attempt_count >= 0),
    created_at          timestamp with time zone default now() not null,
    extend_window_start timestamp with time zone,
    extend_window_count integer                  default 0     not null
);

comment on table email_verifications is '이메일 OTP 인증 기록 (평문 저장 금지)';

comment on column email_verifications.user_id is '회원 FK';

comment on column email_verifications.email is '이메일 조회 편의용';

comment on column email_verifications.code_hash is 'OTP 해시';

comment on column email_verifications.expires_at is '만료 시각(UTC)';

comment on column email_verifications.verified_at is '인증 완료 시각(UTC)';

comment on column email_verifications.attempt_count is '인증 시도 횟수';

comment on column email_verifications.created_at is '생성 시각(UTC)';

alter table email_verifications
    owner to verlnn;

create index idx_ev_user_id
    on email_verifications (user_id);

create index idx_ev_email_created
    on email_verifications (email asc, created_at desc);

create index idx_ev_unverified
    on email_verifications (email)
    where (verified_at IS NULL);
```

- users
```sql
create table users
(
    id            bigint generated by default as identity
        primary key,
    email         varchar(320)
        constraint uk_users_email
            unique,
    password_hash varchar(255),
    tier          varchar(20)                            not null
        references tiers,
    status        varchar(30)                            not null
        constraint chk_users_status
            check ((status)::text = ANY
                   ((ARRAY ['PENDING_VERIFICATION'::character varying, 'ACTIVE'::character varying, 'SUSPENDED'::character varying, 'DELETED'::character varying, 'PENDING_ONBOARDING'::character varying])::text[])),
    phone         varchar(30)
        constraint uk_users_phone
            unique,
    display_name  varchar(50),
    last_login_at timestamp with time zone,
    deleted_at    timestamp with time zone,
    created_at    timestamp with time zone default now() not null,
    updated_at    timestamp with time zone default now() not null
);

comment on table users is '회원 계정 (Tier0 인증/로그인 기준)';

comment on column users.id is '회원 PK';

comment on column users.email is '이메일(로그인 ID)';

comment on column users.password_hash is 'BCrypt 비밀번호 해시';

comment on column users.tier is 'Tier 코드(FK). 권한 SSOT는 코드에 있음';

comment on column users.status is '계정 상태';

comment on column users.phone is '휴대폰 번호(선택)';

comment on column users.display_name is '표시명(선택)';

comment on column users.last_login_at is '마지막 로그인 시각(UTC)';

comment on column users.deleted_at is '소프트 삭제 시각(UTC)';

comment on column users.created_at is '생성 시각(UTC)';

comment on column users.updated_at is '수정 시각(UTC)';
```
